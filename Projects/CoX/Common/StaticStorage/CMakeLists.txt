find_package(Qt5Core REQUIRED)

SET(target_CPP
${CMAKE_CURRENT_SOURCE_DIR}/Filesystem.cpp
${CMAKE_CURRENT_SOURCE_DIR}/CoXHash.cpp
${CMAKE_CURRENT_SOURCE_DIR}/DataStorage.cpp

)
SET(target_INCLUDE
${CMAKE_CURRENT_SOURCE_DIR}/Filesystem.h
${CMAKE_CURRENT_SOURCE_DIR}/DataStorage.h
${CMAKE_CURRENT_SOURCE_DIR}/CoXHash.h
)

SET(target_INCLUDE_DIR ${CMAKE_CURRENT_SOURCE_DIR})
SET(target_DEPENDS "")

SET (target_SOURCES
${target_CPP}
${target_INCLUDE}
)

INCLUDE_DIRECTORIES(${target_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${ACE_path})

#ADD_DEPENDENCIES(static_storage ${target_DEPENDS})

set_property(TARGET static_storage APPEND PROPERTY INCLUDE_DIRECTORIES "$<TARGET_PROPERTY:Qt5::Core,INTERFACE_INCLUDE_DIRECTORIES>")
